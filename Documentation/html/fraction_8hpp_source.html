870 template <typename T> T findRemainder(T first, T second) {
871  if (first < 0)
872  first = -first;
873  if (second < 0)
874  second = -second;
875 
876  // finding mod by repeated subtraction
877  T mod = first;
878  while (mod >= second)
879  mod = mod - second;
880 
881  // for negative numbers
882  if (first < 0)
883  return -mod;
884 
885  return mod;
886 }
