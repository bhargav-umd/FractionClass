853 template <typename T> T findRemainder(T first, T second) {
854  if (first < 0)
855  first = -first;
856  if (second < 0)
857  second = -second;
858 
859  // finding mod by repeated subtraction
860  T mod = first;
861  while (mod >= second)
862  mod = mod - second;
863 
864  return mod;
865 }
